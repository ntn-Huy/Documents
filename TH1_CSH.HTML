Bài 1: Kiểm tra số nguyên tố
Bash Shell (bt1.sh)
#!/bin/bash
read -p "Nhap mot so nguyen: " num
if [ "$num" -lt 2 ]; then
  echo "$num khong phai la so nguyen to."
  exit 0
fi
is_prime=1
for ((i=2; i*i<=num; i++)); do
  if [ $((num % i)) -eq 0 ]; then
    is_prime=0
    break
  fi
done
if [ "$is_prime" -eq 1 ]; then
  echo "$num la so nguyen to."
else
  echo "$num khong phai la so nguyen to."
fi
C Shell (bt1.csh)
#!/bin/csh
echo -n "Nhap mot so nguyen: "
set num = $<
if ( $num < 2 ) then
  echo "$num khong phai la so nguyen to."
  exit
endif
set is_prime = 1
@ i = 2
while ( $i * $i <= $num )
  if ( $num % $i == 0 ) then
    set is_prime = 0
    break
  endif
  @ i++
end
if ( $is_prime == 1 ) then
  echo "$num la so nguyen to."
else
  echo "$num khong phai la so nguyen to."
endif
Bài 2: Liệt kê các số nguyên tố nhỏ hơn n
Bash Shell (bt2.sh)
#!/bin/bash
read -p "Nhap mot so nguyen n: " n
for ((i=2; i<n; i++)); do
  is_prime=1
  for ((j=2; j*j<=i; j++)); do
    if (( i % j == 0 )); then
      is_prime=0
      break
    fi
  done
  if (( is_prime == 1 )); then
    echo "$i"
  fi
done
C Shell (bt2.csh)
#!/bin/csh
echo -n "Nhap mot so nguyen n: "
set n = $<
@ i = 2
while ( $i < $n )
  set is_prime = 1
  @ j = 2
  while ( $j * $j <= $i )
    if ( $i % $j == 0 ) then
      set is_prime = 0
      break
    endif
    @ j++
  end
  if ( $is_prime == 1 ) then
    echo "$i"
  endif
  @ i++
end
Bài 3: Giải phương trình bậc nhất ax + b = 0
Bash Shell (bt3.sh)
#!/bin/bash
read -p "Nhap he so a: " a
read -p "Nhap he so b: " b
if [ "$a" -eq 0 ]; then
  if [ "$b" -eq 0 ]; then
    echo "Phuong trinh vo so nghiem."
  else
    echo "Phuong trinh vo nghiem."
  fi
else
  x=$(echo "scale=2; -$b / $a" | bc)
  echo "Nghiem cua phuong trinh la x = $x"
fi
C Shell (bt3.csh)
#!/bin/csh

set a = $<

  if ( $b == 0 ) then

  endif

  set x = `echo "scale=2; -$b / $a" | bc`

Bài 4: Giải phương trình bậc hai ax² + bx + c = 0
Bash Shell (bt4.sh)
#!/bin/bash
echo "Nhap he so a: "
read a
echo "Nhap he so b: "
read b
echo "Nhap he so c: "
read c
if [ "$a" == "0" ]; then
    # Xu ly pt bac 1...
    exit
fi
delta=$(echo "$b*$b -4*$a*$c" | bc -l )
if (($(echo "$delta < 0" | bc -l)));then
	echo "phuong trinh vo nghiem"
elif (($(echo "$delta == 0" | bc -l)));then
	x=$(echo "scale=2; -($b)/(2*$a)" | bc -l)
	echo "phuong trinh co nghiem kep: x = $x "
else
	sqrt_delta=$(echo "scale =5; sqrt($delta)" | bc -l )
	x1=$(echo "scale=2; (-($b) + $sqrt_delta)/(2*$a)" | bc -l)
	x2=$(echo "scale=2; ( -($b) - $sqrt_delta)/(2*$a)" | bc -l)
	echo "x1 = $x1, x2 = $x2"
fi
C Shell (bt4.csh)
#!/bin/csh
echo -n "Nhap a: "
set a = $<
echo -n "Nhap b: "
set b = $<
echo -n "Nhap c: "
set c = $<
if ($a == 0) then
	echo "Day khong phai pt bac 2 "
	exit 1
endif
set delta = `echo "$b * $b -4 * $a * $c" | bc`
if ( $delta < 0) then
	echo " phuong trinh vo nghiem "
else if ( $delta == 0 ) then
	set x= `echo "scale=2; -($b)/(2* $a)" | bc `
	echo " nghiem kep x = $x"
else
    set sqrtdelta = `echo "scale=5; sqrt($delta)" | bc`
    set x1 = `echo "scale=2; (-($b) + $sqrtdelta) / (2 * $a)" | bc `
    set x2 = `echo "scale=2; (-($b) - $sqrtdelta) / ( 2 * $a) " | bc`
    echo " x1=$x1, x2=$x2"
endif
Bài 5: Tính tổng các số lẻ từ 1 đến n
Bash Shell (bt5.sh)
#!/bin/bash
echo -n " Nhap n: "
read n
tong=0
for (( i=1 ; i<=n ; i++))
do
	if ((i % 2 != 0 ))
	then
		tong=$((tong + i))
	fi
done
echo "Tong cac so le tu 1 den $n la : $tong "
C Shell (bt5.csh)
#!/bin/csh
echo -n "Nhap n: "
set n = $<
set tong=0
set i = 1
while ($i <= $n )
	if ($i % 2 != 0 ) then
	@ tong = $tong + $i
	endif
	@ i= $i + 1
end
echo " tong cac so le la : $tong "
Bài 6: Tính giai thừa của một số
Bash Shell (bt6.sh)
#!/bin/bash
echo "=== Chuong trinh tinh giai thua ==="
read -p "Nhap so n:" n
result=1
for ((i=2;i <= n;i++));do	
	((result *= i ))
done
echo "$n ! = $result"
C Shell (bt6.csh)
#!/bin/csh
echo "Tinh giai thua"
echo "Nhap so: "
set a = $<
set re = 1
set i = 1
while ( $i <= $a )
	@ re *= $i
    @ i++
end
echo "$a!=$re"
Bài 7: Đếm số dòng của tập tin
Bash Shell (bt7.sh)
#!/bin/bash
echo -n "Vui long nhap ten tap tin: "
read ten_file
if [ ! -f "$ten_file" ]; then
    echo "Loi: Tap tin khong ton tai."
    exit 1
fi
so_dong=$(wc -l < "$ten_file")
echo "Tap tin co tong cong: $so_dong dong."
C Shell (bt7.csh)
#!/bin/csh
